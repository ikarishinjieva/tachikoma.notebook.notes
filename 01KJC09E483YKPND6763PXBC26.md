在容器内支持摄像头, 并测试: 

```
sudo mknod /dev/video3 c 81 4
sudo chmod 666 /dev/video3
ros2 run v4l2_camera v4l2_camera_node   --ros-args -p video_device:=/dev/video3 -p image_size:="[1280,480]"
 
在另一个窗口: 
ros2 run image_tools showimage --ros-args -r image:=/image_raw
```

手眼标定: 

```
# 启动slider
python3 slider/slider.launch.py --use-sim-time --enable-apriltag --apriltag-size 0.05 --disable-linkerhand
 
# 启动moveit, 发布机械臂坐标
python3 moveit/moveit.launch.py --use-rviz --use-sim-time
 
# 查看tag_detections
ros2 topic echo tag_detections

# 松开关节
ros2 service call /slider_hw_stack/release_all_servos std_srvs/srv/Trigger "{}"
 
# 记录点位
rm -f eye_to_hand.log; python3 collect_eye_to_hand_calibration.py   --base-frame g_base   --flange-frame joint6_flange   --camera-frame camera_optical   --tag-frame "tag36h11:6" |& tee eye_to_hand.log
  
# 计算结果
python3 collect_eye_to_hand_calibration.py   --calibrate eye_to_hand.log   --base-frame g_base   --flange-frame joint6_flange   --camera-frame camera_optical --trim-percent 20

```
