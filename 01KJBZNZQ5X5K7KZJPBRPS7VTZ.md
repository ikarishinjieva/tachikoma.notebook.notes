# 运行脚本

```
#!/bin/bash
set -e

# phase-1 training
cp /opt/huangyan/ms-swift/swift/llm/sft.py.raw /opt/huangyan/ms-swift/swift/llm/sft.py

bash -c "HTTP_PROXY=http://127.0.0.1:7890 HTTPS_PROXY=http://127.0.0.1:7890 CUDA_VISIBLE_DEVICES=0,1 NPROC_PER_NODE=1  PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
swift sft \
--model_type qwen2-vl-7b-instruct \
--model_id_or_path qwen/Qwen2-VL-7B-Instruct \
--sft_type lora \
--dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/selected_data.length_1_4.500.json \
--val_dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/length_4.ms_swift.val.json \
--eval_steps 50 \
--learning_rate 4e-4 \
--num_train_epochs 10 \
--logging_steps 10 \
" > train_1.log 2>&1

last_model_checkpoint=$(awk -F ': ' '/last_model_checkpoint/ {print $2}' train_1.log)

sleep 10

# merge lora
CUDA_VISIBLE_DEVICES=0 swift export --ckpt_dir "${last_model_checkpoint}" --merge_lora true
last_model_checkpoint="${last_model_checkpoint}-merged"

# phase-2 training
cp /opt/huangyan/ms-swift/swift/llm/sft.py.raw /opt/huangyan/ms-swift/swift/llm/sft.py

bash -c "HTTP_PROXY=http://127.0.0.1:7890 HTTPS_PROXY=http://127.0.0.1:7890 CUDA_VISIBLE_DEVICES=0,1 NPROC_PER_NODE=1  PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
swift sft \
--model_type qwen2-vl-7b-instruct \
--model_id_or_path ${last_model_checkpoint} \
--sft_type lora \
--dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/selected_data.length_5.500.json \
--val_dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/length_5.ms_swift.val.json \
--eval_steps 50 \
--learning_rate 4e-4 \
--num_train_epochs 10 \
--logging_steps 10 \
" > train_2.log 2>&1

last_model_checkpoint=$(awk -F ': ' '/last_model_checkpoint/ {print $2}' train_2.log)

sleep 10

# merge lora
CUDA_VISIBLE_DEVICES=0 swift export --ckpt_dir "${last_model_checkpoint}" --merge_lora true
last_model_checkpoint="${last_model_checkpoint}-merged"

# phase-3 training
cp /opt/huangyan/ms-swift/swift/llm/sft.py.raw /opt/huangyan/ms-swift/swift/llm/sft.py

bash -c "HTTP_PROXY=http://127.0.0.1:7890 HTTPS_PROXY=http://127.0.0.1:7890 CUDA_VISIBLE_DEVICES=0,1 NPROC_PER_NODE=1  PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
swift sft \
--model_type qwen2-vl-7b-instruct \
--model_id_or_path ${last_model_checkpoint} \
--sft_type lora \
--dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/selected_data.length_6.500.json \
--val_dataset /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/length_6.ms_swift.val.json \
--eval_steps 50 \
--learning_rate 4e-4 \
--num_train_epochs 10 \
--logging_steps 10 \
" > train_3.log 2>&1

last_model_checkpoint=$(awk -F ': ' '/last_model_checkpoint/ {print $2}' train_3.log)

sleep 10

## evaluate 
cp /opt/huangyan/ms-swift/swift/llm/sft.py.raw /opt/huangyan/ms-swift/swift/llm/sft.py
 
python3 /opt/huangyan/ms-swift/huangyan_test/evaluate_loop_selfies/evaluate_loop_selfies.py \
    ${last_model_checkpoint} \
    /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/length_6.ms_swift.val.json \
    --limit 300 1>evaluate_1.last_model.log 2>&1

## evaluate 
cp /opt/huangyan/ms-swift/swift/llm/sft.py.raw /opt/huangyan/ms-swift/swift/llm/sft.py
 
python3 /opt/huangyan/ms-swift/huangyan_test/evaluate_loop_selfies/evaluate_loop_selfies.py \
    ${last_model_checkpoint} \
    /opt/huangyan/LLaMA-Factory/data/loop_selfies/image_in_json_out.mol_desc/length_7.ms_swift.val.json \
    --limit 300 1>evaluate_2.last_model.log 2>&1 
```
