在cursor配置中, 增加MCP配置: 

```
{
  "mcpServers": {
    "playwright": {
      "command": "npx",
      "args": [
        "@playwright/mcp@latest"
      ],
      "env": {
        "DISPLAY": ":0"
      }
    }
  }
}
``` 

服务器上安装npm: 

```
export http_proxy=http://10.186.16.135:7890
export https_proxy=http://10.186.16.135:7890
export no_proxy=127.0.0.1,localhost,ucloud.cn
 
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.40.4/install.sh | bash
 
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # This loads nvm
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"  # This loads nvm bash_completion
 
nvm install 20
nvm use 20
 
sudo ln -sf /home/huangyan/.nvm/versions/node/v20.20.0/bin/node /usr/local/bin/node
sudo ln -sf /home/huangyan/.nvm/versions/node/v20.20.0/bin/npm /usr/local/bin/npm
```
